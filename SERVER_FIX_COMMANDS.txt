========================================
  服务器更新命令 - 手动绑定卡片功能
  2026-04-02
========================================

# 1. 进入项目目录拉取最新代码
cd /opt/cardvela
git pull origin main

# 2. 重新构建前端（因为改了页面组件）
npx next build

# 3. 重启应用让后端 API 生效
pm2 restart cardvela --update-env

# 4. 检查是否启动成功
pm2 logs cardvela --nostream --lines 20

# 如果 build 报错，先装一下依赖再 build：
# npm install
# npx next build
