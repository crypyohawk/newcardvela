===== 服务器重新部署（上次git pull失败了）=====

# 1. 强制重置本地改动再拉代码
cd ~/cardvela
git checkout -- .
git clean -fd .next/
git pull

# 2. 确认代码已更新（两行都应该有输出）
grep "force-dynamic" app/api/user/ai-service/keys/route.ts
grep "key-create" app/api/user/ai-service/keys/route.ts

# 3. 构建重启
rm -rf .next
npm install
npm run build
pm2 restart cardvela

# 4. 用户端删除所有旧 key，创建一个新 key

# 5. 查看日志
pm2 logs cardvela --lines 30 --nostream
